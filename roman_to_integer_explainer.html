def romanToInt(s):
    vals={'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
    total=prev=0
    for c in reversed(s):
        v=vals[c]
        total+=v if v>=prev else -v
        prev=v
    return total
